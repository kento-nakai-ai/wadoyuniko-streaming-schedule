### 財務予測モデル

```python
# financial-model.py
import pandas as pd
import numpy as np

class FinancialModel:
    def __init__(self):
        self.assumptions = {
            "initial_customers": 100,
            "monthly_growth_rate": 0.20,  # 20%
            "churn_rate": 0.05,           # 5%
            "average_price": 98000,        # 月額9.8万円
            "gross_margin": 0.85,         # 85%
            "team_growth_rate": 0.15      # 15%/月
        }
    
    def revenue_projection(self, months=36):
        projections = []
        customers = self.assumptions["initial_customers"]
        
        for month in range(1, months + 1):
            # 顧客数計算
            new_customers = customers * self.assumptions["monthly_growth_rate"]
            churned_customers = customers * self.assumptions["churn_rate"]
            customers = customers + new_customers - churned_customers
            
            # 売上計算
            mrr = customers * self.assumptions["average_price"]
            gross_revenue = mrr * self.assumptions["gross_margin"]
            
            projections.append({
                "month": month,
                "customers": int(customers),
                "mrr": int(mrr),
                "gross_revenue": int(gross_revenue),
                "arr": int(mrr * 12)
            })
        
        return pd.DataFrame(projections)
    
    def expense_model(self, months=36):
        expenses = []
        team_size = 5
        
        for month in range(1, months + 1):
            # チームサイズ成長
            if month % 3 == 0:  # 3ヶ月ごとに成長
                team_size *= (1 + self.assumptions["team_growth_rate"])
            
            # 固定費
            salaries = team_size * 600000  # 平均月給60万円
            office_rent = 500000           # 月50万円
            software_tools = 200000        # 月20万円
            
            # 変動費
            customer_count = self.revenue_projection(month).iloc[-1]["customers"]
            infrastructure = customer_count * 1000  # 顧客あたり1000円
            marketing = customer_count * 5000        # 顧客あたり5000円
            
            total_expenses = salaries + office_rent + software_tools + infrastructure + marketing
            
            expenses.append({
                "month": month,
                "team_size": int(team_size),
                "salaries": int(salaries),
                "total_expenses": int(total_expenses)
            })
        
        return pd.DataFrame(expenses)
    
    def profitability_analysis(self, months=36):
        revenue_df = self.revenue_projection(months)
        expense_df = self.expense_model(months)
        
        combined = pd.merge(revenue_df, expense_df, on="month")
        combined["profit"] = combined["gross_revenue"] - combined["total_expenses"]
        combined["profit_margin"] = combined["profit"] / combined["gross_revenue"]
        combined["cumulative_profit"] = combined["profit"].cumsum()
        
        # ブレイクイーブン月を計算
        breakeven_month = combined[combined["cumulative_profit"] > 0].iloc[0]["month"] if len(combined[combined["cumulative_profit"] > 0]) > 0 else None
        
        return combined, breakeven_month
```
